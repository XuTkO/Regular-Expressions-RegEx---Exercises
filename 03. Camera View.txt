import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;
import java.util.Scanner;

public class regex {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        int[] nums = Arrays.stream(scanner.nextLine().split(" "))
                .mapToInt(Integer::parseInt)
                .toArray();

        int deletedElements = nums[0];
        int printedElements = nums[1];

        String[] text = scanner.nextLine().split("\\|");
        List<String> result = new ArrayList<>();


        for (int i = 0; i < text.length; i++) {
            char[]firstChar = text[i].toCharArray();
          
            if (firstChar[0]=='<'){
                result.add(text[i].substring(1));
            }
        }

        char[] exit;
        String exitString = "";

        for (int i = 0; i < result.size(); i++){
            exit=result.get(i).toCharArray();
            if(exit.length > deletedElements+printedElements) {
                for (int a = deletedElements; a < exit.length; a++){
                    exitString+=exit[a];
                }
                result.set(i,exitString);
                exitString="";
            } else {
                for (int a = deletedElements; a < exit.length; a++){
                    exitString+=exit[a];
                }
                result.set(i,exitString);
                exitString="";
            }

        }
       System.out.println(String.join(", ",result));
    }
}
